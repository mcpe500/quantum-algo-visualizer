## Refactor kedua: playback engine

Sekarang banyak komponen punya pola sama:

```txt
currentStep
isPlaying
setInterval
pause
reset
step forward
jump checkpoint
```

Ini muncul di QAOA Hybrid, VQE Quantum Tab, dan CircuitLab. Jangan ulangi lagi di tiap komponen.

Buat:

```ts
export interface PlaybackState {
  currentStep: number;
  isPlaying: boolean;
  totalSteps: number;
}

export type PlaybackAction =
  | { type: 'play' }
  | { type: 'pause' }
  | { type: 'reset' }
  | { type: 'step-forward' }
  | { type: 'jump'; step: number }
  | { type: 'tick' }
  | { type: 'set-total'; totalSteps: number };

export function playbackReducer(
  state: PlaybackState,
  action: PlaybackAction
): PlaybackState {
  switch (action.type) {
    case 'play':
      return state.currentStep >= state.totalSteps - 1
        ? { ...state, currentStep: 0, isPlaying: true }
        : { ...state, isPlaying: true };

    case 'pause':
      return { ...state, isPlaying: false };

    case 'reset':
      return { ...state, currentStep: 0, isPlaying: false };

    case 'step-forward':
      return {
        ...state,
        isPlaying: false,
        currentStep: Math.min(state.currentStep + 1, state.totalSteps - 1),
      };

    case 'jump':
      return {
        ...state,
        isPlaying: false,
        currentStep: clamp(action.step, 0, state.totalSteps - 1),
      };

    case 'tick': {
      const next = Math.min(state.currentStep + 1, state.totalSteps - 1);
      return {
        ...state,
        currentStep: next,
        isPlaying: next < state.totalSteps - 1,
      };
    }

    case 'set-total':
      return {
        ...state,
        totalSteps: action.totalSteps,
        currentStep: Math.min(state.currentStep, action.totalSteps - 1),
      };
  }
}
```

React hook kecil:

```ts
export function usePlaybackController(totalSteps: number, intervalMs: number) {
  const [state, dispatch] = useReducer(playbackReducer, {
    currentStep: 0,
    isPlaying: false,
    totalSteps,
  });

  useEffect(() => {
    dispatch({ type: 'set-total', totalSteps });
  }, [totalSteps]);

  useEffect(() => {
    if (!state.isPlaying) return;

    const timer = window.setInterval(() => {
      dispatch({ type: 'tick' });
    }, intervalMs);

    return () => window.clearInterval(timer);
  }, [state.isPlaying, intervalMs]);

  return { state, dispatch };
}
```

Dengan ini, `QAOAHybridAnimation`, `VQEQuantumTab`, dan `CircuitLabTab` tidak perlu punya timer logic masing-masing.

```ts
export function buildVQEQuantumTabModel(
  result: VQEBenchmarkResult,
  trace: VQETrace | null,
  activeCheckpoint: number,
  animatedIteration: number
): VQEQuantumTabModel {
  const snapshots = result.quantum.iteration_snapshots ?? [];
  const currentSnapshot = snapshots[activeCheckpoint] ?? null;

  const displayEnergy = currentSnapshot
    ? currentSnapshot.energy
    : result.quantum.energy;

  const displayEnergyError = Math.abs(displayEnergy - result.classical.energy);

  return {
    metrics: buildVQEMetrics(result),
    comparison: buildVQEComparison(result),
    convergence: buildConvergenceModel(result, animatedIteration),
    checkpoint: buildCheckpointModel(snapshots, activeCheckpoint),
    energy: {
      value: displayEnergy,
      error: displayEnergyError,
      accuracy: calculateAccuracy(displayEnergy, result.classical.energy),
    },
    parameters: buildParameterModel(result.quantum.optimal_parameters),
    trace,
  };
}
```

```ts
export function buildCircuitPlaybackFrames(
  project: CircuitProject
): CircuitFrame[] {
  const sortedPlacements = sortPlacementsByExecutionOrder(project.placements);

  let state = createInitialStatevector(project.numQubits);
  const frames: CircuitFrame[] = [
    buildInitialCircuitFrame(project, state),
  ];
```
